CREATE MULTISET TABLE PD_TAGCFG.SYS_BASS_USER ,NO FALLBACK ,
     NO BEFORE JOURNAL,
     NO AFTER JOURNAL,
     CHECKSUM = DEFAULT,
     DEFAULT MERGEBLOCKRATIO
     (
      LOGIN_ID VARCHAR(50) CHARACTER SET LATIN NOT CASESPECIFIC TITLE '登录ID' NOT NULL,
      USER_NAME VARCHAR(50) CHARACTER SET LATIN NOT CASESPECIFIC TITLE '登录用户名',
      USER_PHONE VARCHAR(20) CHARACTER SET LATIN CASESPECIFIC TITLE '用户号码',
      LAST_TXDATE TIMESTAMP(0) DEFAULT  CURRENT_TIMESTAMP(0)  TITLE '数据最新日期')
PRIMARY INDEX ( LOGIN_ID );
